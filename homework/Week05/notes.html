yaxs="i", #设置Y轴风格internal

pch=19) #设置散点的形状为实心圆点

  • 增加图例
rain <- read.csv("data/cityrain.csv", header = TRUE)
plot(rain$Tokyo, type = "l", col = "red", ylim = c(0, 300), main = "Monthly Rainfall in major cities", 
    xlab = "Month of Year", ylab = "Rainfall (mm)", lwd = 2)
lines(rain$Tokyo, type = "l", col = "red", lwd = 2)
lines(rain$NewYork, type = "l", col = "blue", lwd = 2)
lines(rain$London, type = "l", col = "green", lwd = 2)
lines(rain$Berlin, type = "l", col = "orange", lwd = 2)
legend("topright", legend = c("Tokyo", "NewYork", "London", "Berlin"), col = c("red", 
    "blue", "green", "orange"), lty = 1, lwd = 2)
legend("top", legend = c("Tokyo", "NewYork", "London", "Berlin"), ncol = 4, 
    cex = 0.8, bty = "n", col = c("red", "blue", "green", "orange"), lty = 1, 
    lwd = 2)
